## 처음 작성 코드
```js
function solution(s) {
    var answer = 0;
    if (s.charAt(0)==='+'){
        s = s.slice(1,s.length)
        answer = parseInt(s)
    }else if (s.charAt(0)==='-'){
        s=s.slice(1,s.length)
        answer = parseInt(s) * -1
    }else{
        answer=parseInt(s)
    }
    return answer;
}
```
